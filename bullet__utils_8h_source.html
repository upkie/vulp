63 inline int find_link_index(b3RobotSimulatorClientAPI& bullet, int robot,
64  const std::string& link_name) noexcept {
65  b3JointInfo joint_info;
66  int nb_joints = bullet.getNumJoints(robot);
67  for (int joint_index = 0; joint_index < nb_joints; ++joint_index) {
68  bullet.getJointInfo(robot, joint_index, &joint_info);
69  if (std::string(joint_info.m_linkName) == link_name) {
70  return joint_index; // link and joint indices are the same in Bullet
71  }
72  }
73  return -1;
74 }

107 inline void read_imu_data(BulletImuData& imu_data,
108  b3RobotSimulatorClientAPI& bullet, int robot,
109  const int imu_link_index, double dt) {
110  b3LinkState link_state;
111  bullet.getLinkState(robot, imu_link_index, /* computeVelocity = */ true,
112  /* computeForwardKinematics = */ true, &link_state);
113 
114  Eigen::Quaterniond orientation_imu_in_world;
115  orientation_imu_in_world.w() = link_state.m_worldLinkFrameOrientation[3];
116  orientation_imu_in_world.x() = link_state.m_worldLinkFrameOrientation[0];
117  orientation_imu_in_world.y() = link_state.m_worldLinkFrameOrientation[1];
118  orientation_imu_in_world.z() = link_state.m_worldLinkFrameOrientation[2];
119 
120  // The attitude reference system frame has +x forward, +y right and +z down,
121  // whereas our world frame has +x forward, +y left and +z up:
122  // https://github.com/mjbots/pi3hat/blob/master/docs/reference.md#orientation
123  Eigen::Matrix3d rotation_world_to_ars =
124  Eigen::Vector3d{1.0, -1.0, -1.0}.asDiagonal();
125 
126  Eigen::Matrix3d rotation_imu_to_world =
127  orientation_imu_in_world.toRotationMatrix();
128  Eigen::Matrix3d rotation_imu_to_ars =
129  rotation_world_to_ars * rotation_imu_to_world;
130  Eigen::Quaterniond orientation_imu_in_ars(rotation_imu_to_ars);
131 
132  Eigen::Vector3d linear_velocity_imu_in_world = {
133  link_state.m_worldLinearVelocity[0],
134  link_state.m_worldLinearVelocity[1],
135  link_state.m_worldLinearVelocity[2],
136  };
137 
138  Eigen::Vector3d angular_velocity_imu_to_world_in_world = {
139  link_state.m_worldAngularVelocity[0],
140  link_state.m_worldAngularVelocity[1],
141  link_state.m_worldAngularVelocity[2],
142  };
143 
144  // Compute linear acceleration in the world frame by discrete differentiation
145  const auto& previous_linear_velocity = imu_data.linear_velocity_imu_in_world;
146  Eigen::Vector3d linear_acceleration_imu_in_world =
147  (linear_velocity_imu_in_world - previous_linear_velocity) / dt;
148 
149  auto rotation_world_to_imu = orientation_imu_in_world.normalized().inverse();
150  Eigen::Vector3d angular_velocity_imu_in_imu =
151  rotation_world_to_imu * angular_velocity_imu_to_world_in_world;
152  Eigen::Vector3d linear_acceleration_imu_in_imu =
153  rotation_world_to_imu * linear_acceleration_imu_in_world;
154 
155  // Fill out regular IMU data
156  imu_data.orientation_imu_in_ars = orientation_imu_in_ars;
157  imu_data.angular_velocity_imu_in_imu = angular_velocity_imu_in_imu;
158  imu_data.linear_acceleration_imu_in_imu = linear_acceleration_imu_in_imu;
159 
160  // ... and the extra field for the Bullet interface
161  imu_data.linear_velocity_imu_in_world = linear_velocity_imu_in_world;
162 }

171 double compute_robot_mass(b3RobotSimulatorClientAPI& bullet, int robot) {
172  b3DynamicsInfo dynamics_info;
173  double mass = 0; // kg
174  const int nb_joints = bullet.getNumJoints(robot);
175  for (int joint_index = 0; joint_index < nb_joints; ++joint_index) {
176  bullet.getDynamicsInfo(robot, joint_index, &dynamics_info);
177  mass += dynamics_info.m_mass;
178  }
179  return mass;
180 }

191 Eigen::Vector3d compute_position_com_in_world(b3RobotSimulatorClientAPI& bullet,
192  int robot) {
193  b3LinkState link_state;
194  b3DynamicsInfo dynamics_info;
195  double mass = 0.0; // kg
196  Eigen::Vector3d weighted_sum = Eigen::Vector3d::Zero();
197  const int nb_joints = bullet.getNumJoints(robot);
198  for (int link_index = 0; link_index < nb_joints; ++link_index) {
199  bullet.getDynamicsInfo(robot, link_index, &dynamics_info);
200  const double link_mass = dynamics_info.m_mass;
201  Eigen::Vector3d position_link_com_in_link = {
202  dynamics_info.m_localInertialFrame[0],
203  dynamics_info.m_localInertialFrame[1],
204  dynamics_info.m_localInertialFrame[2],
205  };
206 
207  bullet.getLinkState(robot, link_index, /* computeVelocity = */ false,
208  /* computeForwardKinematics = */ true, &link_state);
209  Eigen::Quaterniond orientation_link_in_world;
210  orientation_link_in_world.w() = link_state.m_worldLinkFrameOrientation[3];
211  orientation_link_in_world.x() = link_state.m_worldLinkFrameOrientation[0];
212  orientation_link_in_world.y() = link_state.m_worldLinkFrameOrientation[1];
213  orientation_link_in_world.z() = link_state.m_worldLinkFrameOrientation[2];
214  Eigen::Vector3d position_link_in_world = {
215  link_state.m_worldLinkFramePosition[0],
216  link_state.m_worldLinkFramePosition[1],
217  link_state.m_worldLinkFramePosition[2],
218  };
219  Eigen::Matrix3d rotation_link_to_world =
220  orientation_link_in_world.toRotationMatrix();
221  Eigen::Vector3d position_link_com_in_world =
222  rotation_link_to_world * position_link_com_in_link +
223  position_link_in_world;
224 
225  weighted_sum += link_mass * position_link_com_in_world;
226  mass += link_mass;
227  }
228  return weighted_sum / mass;
229 }
