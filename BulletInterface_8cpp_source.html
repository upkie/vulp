27 BulletInterface::BulletInterface(const ServoLayout& layout,
28  const Parameters& params)
29  : Interface(layout), params_(params) {
30  // Start simulator
31  auto flag = (params.gui ? eCONNECT_GUI : eCONNECT_DIRECT);
32  bool is_connected = bullet_.connect(flag);
33  if (!is_connected) {
34  throw std::runtime_error("Could not connect to the Bullet GUI");
35  }
36 
37  // Setup simulation scene
38  bullet_.configureDebugVisualizer(COV_ENABLE_GUI, 0);
39  bullet_.configureDebugVisualizer(COV_ENABLE_RENDERING, 0);
40  bullet_.configureDebugVisualizer(COV_ENABLE_SHADOWS, 0);
41  if (params.gravity) {
42  bullet_.setGravity(btVector3(0, 0, -9.81));
43  }
44  bullet_.setRealTimeSimulation(false); // making sure
45 
46  // Load robot model
47  robot_ = bullet_.loadURDF(params.robot_urdf_path);
48  imu_link_index_ = get_link_index("imu");
49  if (imu_link_index_ < 0) {
50  throw std::runtime_error("Robot does not have a link named \"imu\"");
51  }
52 
53  // Read servo layout
54  for (const auto& id_joint : servo_joint_map()) {
55  const auto& servo_id = id_joint.first;
56  const std::string& joint_name = id_joint.second;
57  moteus::ServoReply reply;
58  reply.id = servo_id;
59  reply.result.temperature = 20.0; // ['C], simulation room temperature
60  reply.result.voltage = 18.0; // [V], nominal voltage of a RYOBI battery
61  joint_index_map_.try_emplace(joint_name, -1);
62  servo_reply_.try_emplace(joint_name, reply);
63  }
64 
65  // Map servo layout to Bullet
66  b3JointInfo joint_info;
67  const int nb_joints = bullet_.getNumJoints(robot_);
68  for (int joint_index = 0; joint_index < nb_joints; ++joint_index) {
69  bullet_.getJointInfo(robot_, joint_index, &joint_info);
70  std::string joint_name = joint_info.m_jointName;
71  if (joint_index_map_.find(joint_name) != joint_index_map_.end()) {
72  joint_index_map_[joint_name] = joint_index;
73 
74  BulletJointProperties props;
75  props.maximum_torque = joint_info.m_jointMaxForce;
76  joint_properties_.try_emplace(joint_name, props);
77  }
78  }
79 
80  // Load plane URDF
81  if (params.floor) {
82  if (bullet_.loadURDF(find_plane_urdf(params.argv0)) < 0) {
83  throw std::runtime_error("Could not load the plane URDF!");
84  }
85  } else {
86  spdlog::info("Not loading the plane URDF");
87  if (params.gravity) {
88  spdlog::warn(
89  "No ground plane was loaded, but gravity is enabled. The robot will "
90  "fall!");
91  }
92  }
93 
94  // Load environment URDFs
95  for (const auto& urdf_path : params.env_urdf_paths) {
96  spdlog::info("Loading environment URDF: ", urdf_path);
97  if (bullet_.loadURDF(urdf_path) < 0) {
98  throw std::runtime_error("Could not load the environment URDF: " +
99  urdf_path);
100  }
101  }
102 
103  // Start visualizer and configure simulation
104  bullet_.configureDebugVisualizer(COV_ENABLE_RENDERING, 1);
105  reset(Dictionary{});
106 }

122 void BulletInterface::reset_base_state(
123  const Eigen::Vector3d& position_base_in_world,
124  const Eigen::Quaterniond& orientation_base_in_world,
125  const Eigen::Vector3d& linear_velocity_base_to_world_in_world,
126  const Eigen::Vector3d& angular_velocity_base_in_base) {
127  bullet_.resetBasePositionAndOrientation(
128  robot_, bullet_from_eigen(position_base_in_world),
129  bullet_from_eigen(orientation_base_in_world));
130 
131  const auto& rotation_base_to_world =
132  orientation_base_in_world; // transforms are coordinates
133  const Eigen::Vector3d angular_velocity_base_in_world =
134  rotation_base_to_world * angular_velocity_base_in_base;
135  bullet_.resetBaseVelocity(
136  robot_, bullet_from_eigen(linear_velocity_base_to_world_in_world),
137  bullet_from_eigen(angular_velocity_base_in_world));

154  b3JointInfo joint_info;
155  const int nb_joints = bullet_.getNumJoints(robot_);
156  for (int joint_index = 0; joint_index < nb_joints; ++joint_index) {
157  bullet_.getJointInfo(robot_, joint_index, &joint_info);
158  std::string joint_name = joint_info.m_jointName;
159  if (joint_index_map_.find(joint_name) != joint_index_map_.end()) {

204 void BulletInterface::process_forces(const Dictionary& external_forces) {
205  for (const auto& link_name : external_forces.keys()) {
206  int link_index = get_link_index(link_name);
207  if (link_index < 0) {
208  if (link_name == "base") {
209  link_index = -1;
210  } else {
211  spdlog::warn(
212  "Link \"{}\" not found in the robot description, cannot apply an "
213  "external force to it",
214  link_name);
215  continue;
216  }
217  }
218  const auto& params = external_forces(link_name);
219  const bool local_frame = params.get<bool>("local_frame", false);
220  Eigen::Vector3d position_eigen;
221  int flags;
222  if (local_frame) {
223  flags = EF_LINK_FRAME;
224  position_eigen.setZero();
225  } else /* world frame */ {
226  flags = EF_WORLD_FRAME;
227  position_eigen = vulp::actuation::get_position_link_in_world(
228  bullet_, robot_, link_index);
229  }
230  btVector3 position = bullet_from_eigen(position_eigen);
231  btVector3 force = bullet_from_eigen(params.get<Eigen::Vector3d>("force"));
232  bullet_.applyExternalForce(robot_, link_index, force, position, flags);
233  }
234 }

236 void BulletInterface::cycle(
237  std::function<void(const moteus::Output&)> callback) {
238  const moteus::Data& data = this->data_;
239  assert(data.commands.size() == data.replies.size());
240  assert(!std::isnan(params_.dt));
241  if (!bullet_.isConnected()) {
242  throw std::runtime_error("simulator is not running any more");
243  }
244 
245  read_joint_sensors();
246  read_imu_data(imu_data_, bullet_, robot_, imu_link_index_, params_.dt);
247  read_contacts();
248  send_commands(data);
249  bullet_.stepSimulation();
250 
251  if (params_.follower_camera) {
252  translate_camera_to_robot();
253  }
254 
255  moteus::Output output;
256  for (size_t i = 0; i < data.replies.size(); ++i) {
257  const auto servo_id = data.commands[i].id;
258  const std::string& joint_name = servo_layout().joint_name(servo_id);
259  data.replies[i].id = servo_id;
260  data.replies[i].result = servo_reply_[joint_name].result;
261  output.query_result_size = i + 1;
262  }
263  callback(output);
264 }

278 void BulletInterface::read_joint_sensors() {
279  b3JointSensorState sensor_state;
280  for (const auto& name_index : joint_index_map_) {
281  const auto& joint_name = name_index.first;
282  const auto joint_index = name_index.second;
283  bullet_.getJointState(robot_, joint_index, &sensor_state);
284  auto& result = servo_reply_[joint_name].result;
285  result.position = sensor_state.m_jointPosition / (2.0 * M_PI);
286  result.velocity = sensor_state.m_jointVelocity / (2.0 * M_PI);
287  result.torque = sensor_state.m_jointMotorTorque;
288  }

291 void BulletInterface::send_commands(const moteus::Data& data) {
292  b3RobotSimulatorJointMotorArgs motor_args(CONTROL_MODE_VELOCITY);
293 
294  for (const auto& command : data.commands) {
295  const auto servo_id = command.id;
296  const std::string& joint_name = servo_layout().joint_name(servo_id);
297  const int joint_index = joint_index_map_[joint_name];
298 
299  const auto previous_mode = servo_reply_[joint_name].result.mode;
300  if (previous_mode == moteus::Mode::kStopped &&
301  command.mode != moteus::Mode::kStopped) {
302  // disable velocity controllers to enable torque control
303  motor_args.m_controlMode = CONTROL_MODE_VELOCITY;
304  motor_args.m_maxTorqueValue = 0.; // [N m]
305  bullet_.setJointMotorControl(robot_, joint_index, motor_args);
306  }
307  servo_reply_[joint_name].result.mode = command.mode;
308 
309  if (command.mode == moteus::Mode::kStopped) {
310  motor_args.m_controlMode = CONTROL_MODE_VELOCITY;
311  motor_args.m_maxTorqueValue = 100.; // [N m]
312  motor_args.m_targetVelocity = 0.; // [rad] / [s]
313  bullet_.setJointMotorControl(robot_, joint_index, motor_args);
314  continue;
315  }
316 
317  if (command.mode != moteus::Mode::kPosition) {
318  throw std::runtime_error(
319  "Bullet interface does not support command mode " +
320  std::to_string(static_cast<unsigned>(command.mode)));
321  }
322 
323  const double target_position = command.position.position * (2.0 * M_PI);
324  const double target_velocity = command.position.velocity * (2.0 * M_PI);
325  const double feedforward_torque = command.position.feedforward_torque;
326  const double kp_scale = command.position.kp_scale;
327  const double kd_scale = command.position.kd_scale;
328  const double maximum_torque = command.position.maximum_torque;
329  const double joint_torque = compute_joint_torque(
330  joint_name, feedforward_torque, target_position, target_velocity,
331  kp_scale, kd_scale, maximum_torque);
332  motor_args.m_controlMode = CONTROL_MODE_TORQUE;
333  motor_args.m_maxTorqueValue = joint_torque;
334  servo_reply_[joint_name].result.torque = joint_torque;
335  bullet_.setJointMotorControl(robot_, joint_index, motor_args);
336  }
337 }

339 double BulletInterface::compute_joint_torque(
340  const std::string& joint_name, const double feedforward_torque,
341  const double target_position, const double target_velocity,
342  const double kp_scale, const double kd_scale, const double maximum_torque) {
343  assert(!std::isnan(target_velocity));
344  const BulletJointProperties& joint_props = joint_properties_[joint_name];
345  const auto& measurements = servo_reply_[joint_name].result;
346  const double measured_position = measurements.position * (2.0 * M_PI);
347  const double measured_velocity = measurements.velocity * (2.0 * M_PI);
348  const double kp = kp_scale * params_.torque_control_kp;
349  const double kd = kd_scale * params_.torque_control_kd;
350  const double tau_max = std::min(maximum_torque, joint_props.maximum_torque);
351  double torque = feedforward_torque;
352  torque += kd * (target_velocity - measured_velocity);
353  if (!std::isnan(target_position)) {
354  torque += kp * (target_position - measured_position);
355  }
356  constexpr double kMaxStictionVelocity = 1e-3; // rad/s
357  if (std::abs(measured_velocity) > kMaxStictionVelocity) {
358  torque += joint_props.friction * ((measured_velocity > 0.0) ? -1.0 : +1.0);
359  }
360  torque = std::max(std::min(torque, tau_max), -tau_max);
361  return torque;
362 }

364 Eigen::Matrix4d BulletInterface::transform_base_to_world() const noexcept {
365  btVector3 position_base_in_world;
366  btQuaternion orientation_base_in_world;
367  bullet_.getBasePositionAndOrientation(robot_, position_base_in_world,
368  orientation_base_in_world);
369  auto quat = eigen_from_bullet(orientation_base_in_world);
370  Eigen::Matrix4d T = Eigen::Matrix4d::Identity();
371  T.block<3, 3>(0, 0) = quat.normalized().toRotationMatrix();
372  T.block<3, 1>(0, 3) = eigen_from_bullet(position_base_in_world);
373  return T;
374 }
