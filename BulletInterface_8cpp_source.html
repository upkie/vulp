27 BulletInterface::BulletInterface(const ServoLayout& layout,
28  const Parameters& params)
29  : Interface(layout), params_(params) {
30  // Start simulator
31  auto flag = (params.gui ? eCONNECT_GUI : eCONNECT_DIRECT);
32  bool is_connected = bullet_.connect(flag);
33  if (!is_connected) {
34  throw std::runtime_error("Could not connect to the Bullet GUI");
35  }
36 
37  // Setup simulation scene
38  bullet_.configureDebugVisualizer(COV_ENABLE_GUI, 0);
39  bullet_.configureDebugVisualizer(COV_ENABLE_RENDERING, 0);
40  bullet_.configureDebugVisualizer(COV_ENABLE_SHADOWS, 0);
41  if (params.gravity) {
42  bullet_.setGravity(btVector3(0, 0, -9.81));
43  }
44  bullet_.setRealTimeSimulation(false); // making sure
45 
46  // Load robot model
47  robot_ = bullet_.loadURDF(params.robot_urdf_path);
48  imu_link_index_ = find_link_index(bullet_, robot_, "imu");
49  if (imu_link_index_ < 0) {
50  throw std::runtime_error("Robot does not have a link named \"imu\"");
51  }
52 
53  // Read servo layout
54  for (const auto& id_joint : servo_joint_map()) {
55  const auto& servo_id = id_joint.first;
56  const std::string& joint_name = id_joint.second;
57  moteus::ServoReply reply;
58  reply.id = servo_id;
59  reply.result.temperature = 20.0; // ['C], simulation room temperature
60  reply.result.voltage = 18.0; // [V], nominal voltage of a RYOBI battery
61  joint_index_map_.try_emplace(joint_name, -1);
62  servo_reply_.try_emplace(joint_name, reply);
63  }
64 
65  // Map servo layout to Bullet
66  b3JointInfo joint_info;
67  const int nb_joints = bullet_.getNumJoints(robot_);
68  for (int joint_index = 0; joint_index < nb_joints; ++joint_index) {
69  bullet_.getJointInfo(robot_, joint_index, &joint_info);
70  std::string joint_name = joint_info.m_jointName;
71  if (joint_index_map_.find(joint_name) != joint_index_map_.end()) {
72  joint_index_map_[joint_name] = joint_index;
73 
74  BulletJointProperties props;
75  props.maximum_torque = joint_info.m_jointMaxForce;
76  joint_properties_.try_emplace(joint_name, props);
77  }
78  }
79 
80  // Load plane URDF
81  if (params.floor) {
82  if (bullet_.loadURDF(find_plane_urdf(params.argv0)) < 0) {
83  throw std::runtime_error("Could not load the plane URDF!");
84  }
85  } else {
86  spdlog::info("Not loading the plane URDF");
87  if (params.gravity) {
88  spdlog::warn(
89  "No ground plane was loaded, but gravity is enabled. The robot will "
90  "fall!");
91  }
92  }
93 
94  // Load environment URDFs
95  for (const auto& urdf_path : params.env_urdf_paths) {
96  spdlog::info("Loading environment URDF: ", urdf_path);
97  if (bullet_.loadURDF(urdf_path) < 0) {
98  throw std::runtime_error("Could not load the environment URDF: " +
99  urdf_path);
100  }
101  }
102 
103  // Start visualizer and configure simulation
104  bullet_.configureDebugVisualizer(COV_ENABLE_RENDERING, 1);
105  reset(Dictionary{});
106 }

121 void BulletInterface::reset_base_state(
122  const Eigen::Vector3d& position_base_in_world,
123  const Eigen::Quaterniond& orientation_base_in_world,
124  const Eigen::Vector3d& linear_velocity_base_to_world_in_world,
125  const Eigen::Vector3d& angular_velocity_base_in_base) {
126  bullet_.resetBasePositionAndOrientation(
127  robot_, bullet_from_eigen(position_base_in_world),
128  bullet_from_eigen(orientation_base_in_world));
129 
130  const auto& rotation_base_to_world =
131  orientation_base_in_world; // transforms are coordinates
132  const Eigen::Vector3d angular_velocity_base_in_world =
133  rotation_base_to_world * angular_velocity_base_in_base;
134  bullet_.resetBaseVelocity(
135  robot_, bullet_from_eigen(linear_velocity_base_to_world_in_world),
136  bullet_from_eigen(angular_velocity_base_in_world));
137 }

146 void BulletInterface::reset_joint_properties() {
147  b3JointInfo joint_info;
148  const int nb_joints = bullet_.getNumJoints(robot_);
149  for (int joint_index = 0; joint_index < nb_joints; ++joint_index) {
150  bullet_.getJointInfo(robot_, joint_index, &joint_info);
151  std::string joint_name = joint_info.m_jointName;
152  if (joint_index_map_.find(joint_name) != joint_index_map_.end()) {
153  const auto friction_it = params_.joint_friction.find(joint_name);
154  if (friction_it != params_.joint_friction.end()) {
155  joint_properties_[joint_name].friction = friction_it->second;
156  } else {
157  joint_properties_[joint_name].friction = 0.0;
158  }
159  }
160  }

163 void BulletInterface::cycle(
164  const moteus::Data& data,
165  std::function<void(const moteus::Output&)> callback) {
166  assert(data.commands.size() == data.replies.size());
167  assert(!std::isnan(params_.dt));
168 
169  read_joint_sensors();
170  read_imu_data(imu_data_, bullet_, robot_, imu_link_index_, params_.dt);
171  send_commands(data);
172  bullet_.stepSimulation();
173 
174  if (params_.follower_camera) {
175  translate_camera_to_robot();
176  }
177 
178  moteus::Output output;
179  for (size_t i = 0; i < data.replies.size(); ++i) {
180  const auto servo_id = data.commands[i].id;
181  const std::string& joint_name = servo_layout().joint_name(servo_id);
182  data.replies[i].id = servo_id;
183  data.replies[i].result = servo_reply_[joint_name].result;
184  output.query_result_size = i + 1;
185  }
186  callback(output);
187 }

189 void BulletInterface::read_joint_sensors() {
190  b3JointSensorState sensor_state;
191  for (const auto& name_index : joint_index_map_) {
192  const auto& joint_name = name_index.first;
193  const auto joint_index = name_index.second;
194  bullet_.getJointState(robot_, joint_index, &sensor_state);
195  auto& result = servo_reply_[joint_name].result;
196  result.position = sensor_state.m_jointPosition / (2.0 * M_PI);
197  result.velocity = sensor_state.m_jointVelocity / (2.0 * M_PI);
198  result.torque = sensor_state.m_jointMotorTorque;
199  }
200 }

202 void BulletInterface::send_commands(const moteus::Data& data) {
203  b3RobotSimulatorJointMotorArgs motor_args(CONTROL_MODE_VELOCITY);
204 
205  for (const auto& command : data.commands) {
206  const auto servo_id = command.id;
207  const std::string& joint_name = servo_layout().joint_name(servo_id);
208  const int joint_index = joint_index_map_[joint_name];
209 
210  const auto previous_mode = servo_reply_[joint_name].result.mode;
211  if (previous_mode == moteus::Mode::kStopped &&
212  command.mode != moteus::Mode::kStopped) {
213  // disable velocity controllers to enable torque control
214  motor_args.m_controlMode = CONTROL_MODE_VELOCITY;
215  motor_args.m_maxTorqueValue = 0.; // [N m]
216  bullet_.setJointMotorControl(robot_, joint_index, motor_args);
217  }
218  servo_reply_[joint_name].result.mode = command.mode;
219 
220  if (command.mode == moteus::Mode::kStopped) {
221  motor_args.m_controlMode = CONTROL_MODE_VELOCITY;
222  motor_args.m_maxTorqueValue = 100.; // [N m]
223  motor_args.m_targetVelocity = 0.; // [rad] / [s]
224  bullet_.setJointMotorControl(robot_, joint_index, motor_args);
225  continue;
226  }
227 
228  if (command.mode != moteus::Mode::kPosition) {
229  throw std::runtime_error(
230  "Bullet interface does not support command mode " +
231  std::to_string(static_cast<unsigned>(command.mode)));
232  }
233 
234  // TODO(scaron): introduce control_position/velocity intermediates
235  // See https://github.com/mjbots/moteus/blob/main/docs/reference.md
236  const double target_position = command.position.position * (2.0 * M_PI);
237  const double target_velocity = command.position.velocity * (2.0 * M_PI);
238  const double feedforward_torque = command.position.feedforward_torque;
239  const double kp_scale = command.position.kp_scale;
240  const double kd_scale = command.position.kd_scale;
241  const double maximum_torque = command.position.maximum_torque;
242  const double joint_torque = compute_joint_torque(
243  joint_name, feedforward_torque, target_position, target_velocity,
244  kp_scale, kd_scale, maximum_torque);
245  motor_args.m_controlMode = CONTROL_MODE_TORQUE;
246  motor_args.m_maxTorqueValue = joint_torque;
247  servo_reply_[joint_name].result.torque = joint_torque;
248  bullet_.setJointMotorControl(robot_, joint_index, motor_args);
249  }
250 }

252 double BulletInterface::compute_joint_torque(
253  const std::string& joint_name, const double feedforward_torque,
254  const double target_position, const double target_velocity,
255  const double kp_scale, const double kd_scale, const double maximum_torque) {
256  assert(!std::isnan(target_velocity));
257  const BulletJointProperties& joint_props = joint_properties_[joint_name];
258  const auto& measurements = servo_reply_[joint_name].result;
259  const double measured_position = measurements.position * (2.0 * M_PI);
260  const double measured_velocity = measurements.velocity * (2.0 * M_PI);
261  const double kp = kp_scale * params_.torque_control_kp;
262  const double kd = kd_scale * params_.torque_control_kd;
263  const double tau_max = std::min(maximum_torque, joint_props.maximum_torque);
264  double torque = feedforward_torque;
265  torque += kd * (target_velocity - measured_velocity);
266  if (!std::isnan(target_position)) {
267  torque += kp * (target_position - measured_position);
268  }
269  constexpr double kMaxStictionVelocity = 1e-3; // rad/s
270  if (std::abs(measured_velocity) > kMaxStictionVelocity) {
271  torque += joint_props.friction * ((measured_velocity > 0.0) ? -1.0 : +1.0);
272  }
273  torque = std::max(std::min(torque, tau_max), -tau_max);
274  return torque;
275 }

277 Eigen::Matrix4d BulletInterface::transform_base_to_world() const noexcept {
278  btVector3 position_base_in_world;
279  btQuaternion orientation_base_in_world;
280  bullet_.getBasePositionAndOrientation(robot_, position_base_in_world,
281  orientation_base_in_world);
282  auto quat = eigen_from_bullet(orientation_base_in_world);
283  Eigen::Matrix4d T = Eigen::Matrix4d::Identity();
284  T.block<3, 3>(0, 0) = quat.normalized().toRotationMatrix();
285  T.block<3, 1>(0, 3) = eigen_from_bullet(position_base_in_world);
286  return T;
287 }

297 Eigen::Vector3d BulletInterface::angular_velocity_base_in_base()
298  const noexcept {
299  btVector3 angular_velocity_base_to_world_in_world;
300  btVector3 _; // anonymous, we only get the angular velocity
301  bullet_.getBaseVelocity(robot_, _, angular_velocity_base_to_world_in_world);
302  Eigen::Matrix4d T = transform_base_to_world();
303  Eigen::Matrix3d rotation_base_to_world = T.block<3, 3>(0, 0);
304  return rotation_base_to_world.transpose() *
305  eigen_from_bullet(angular_velocity_base_to_world_in_world);
306 }
